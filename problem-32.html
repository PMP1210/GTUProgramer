Problem statement:-


Write a program to print following patterns:
1
12
123
1234
12345

Code:-


#include <stdio.h>
int main()
{
printf("_____________________________________");
printf("\n\tPatterns of numbers");
printf("\n_____________________________________\n");

int i,j;
for(i=1;i<=5;i++)
{
for(j=1;j< =i;j++)
{
printf("%d",j);
}
printf("\n");
}
printf("\n\n");
return 0;
}
