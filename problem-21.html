Problem 21

Problem statement:-


Write a C program to find the sum and average of different numbers which are accepted by user as many as user wants.

Code:-


#include <stdio.h>
int main()
{
printf("\n___________________________________________");
printf("\n\tFIND SUM,AVG OF MANY NUMBERS");
printf("\n___________________________________________");

int i,a[100],b,sum=0;
float avg=0;
printf("\nEnter the number you want's to add:");
scanf("%d",&b);

for(i=0;i< b;i++)
{
scanf("%d",&a[i]);
sum=sum+a[i];
avg=(sum)/b;
}

printf("\nThe sum is =%d",sum);
printf("\nThe average is =%.1f",avg);

return 0;
}
