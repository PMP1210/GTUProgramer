Problem 26

Problem statement:-


Write a C program to find 1+1/2+1/3+1/4+.+1/n.

Code:-


#include <stdio.h>
int main()
{
printf("\n___________________________________________________");
printf("\n\tFIND 1+1/2+1/3+1+....+1/n");
printf("\n___________________________________________________");

float i,n;
float d=0;
printf("\nEnter the n value:");
scanf("%f",&n);

for(i=1;i<=n;i++)
{
printf("%.1f + ",1/i);
d+=1/i;
}
printf("\nTotal of N number's=%.1f",d);
return 0;
}
