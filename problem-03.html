Problem 03

Problem statement:-


Write a program to find the simple interest. (i=(p*r*n)/100) i=Simple interest , p=Principal amount , r=Rate of interest , n=Number of years

Code:-


#include <stdio.h>
int main()
{
printf("\n_______________________________");
printf("\n\tSIMPLE INTREST");
printf("\n_______________________________");
int p,r;
float n,i;
printf("\n\nEnter the price value:");
scanf("%d",&p);
printf("\n\nEnter the rate value:");
scanf("%d",&r);

printf("\n\nEnter the number of years:");
scanf("%f",&n);

i=(p*r*n)/100;
printf("\n\nSimple intrest i= %f",i);
return 0;
}
