Problem 05

Problem statement:-


Write a C program to enter a distance in to kilometer and convert it in to meter, feet, inches and centimeter

Code:-


#include stdio.h
int main()
{
printf("_______________________________________");
printf("\n\tCONVERT KILOMETERS INTO OTHER ");
printf("\n_______________________________________");

float kilometer,meter,feet,inches,centimeter;
printf("\nEnter the distance in kilometer:");
scanf("%f",&kilometer);

meter=kilometer*1000;
printf("\n\nThe distance in meter is=%.2f",meter);

feet=kilometer*3281;
printf("\n\nThe distance in feet is=%.2f",feet);

inches=kilometer*3281*12;
printf("\n\nThe distance in inches is=%.2f",inches);

centimeter=kilometer*10000;
printf("\n\nThe distance in centimeter is=%.2f",centimeter);

return 0;
}
