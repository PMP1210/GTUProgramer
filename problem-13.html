Problem statement:-


Write a c program to prepare pay slip using following data. Da = 10% of basic, Hra = 7.50% of basic, Ma = 300, Pf = 12.50% of basic, Gross = basic + Da + Hra + Ma, Nt = Gross â€“ Pf.

Code:-


#include <stdio.h>
int main()
{
printf("\n___________________________________________");
printf("\n\n\tPAY - SLIP");
printf("\n___________________________________________");

float a,da=0,hra=0,ma=300,pf=0,gross=0,nt=0;
printf("\nEnter the basic amount :");
scanf("%f",&a);

da=(a*10)/100;
hra=(a*7.50)/100;
pf=(a*12.50)/100;
gross=a+da+hra+ma;
nt=gross-pf;

printf("\n\nThe da of %.1f is =%.1f",a,da);
printf("\n\nThe hra of %.1f is =%.1f",a,hra);
printf("\n\nThe pf of %.1f is =%.1f",a,pf);
printf("\n\nThe gross of %.1f is =%.1f",a,gross);
printf("\n\nThe nt of %.1f is =%.1f",a,nt);

return 0;
}
