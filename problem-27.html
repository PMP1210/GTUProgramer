Problem 27

Problem statement:-


Write a C program to find 1+1/2!+1/3!+1/4!+. +1/n!.

Code:-


#include <stdio.h>
#include <Math.h>
int main()
{
printf("___________________________________________________");
printf("\n\tFIND 1+1/2!+1/3!+1/4!+....+1/n!");
printf("\n___________________________________________________");

int i,n,j;
float sum=0,fac=1;
printf("\nEnter the n value:");
scanf("%d",&n);

for(i=1;i<=n;i++)
{
fac=fac*i;
sum=sum+(1.0/fac);
}
printf("\nTotal of N number's=%.2f",sum);
return 0;
}
