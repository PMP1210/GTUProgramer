Problem statement:-


Write a program that defines a function to add first n numbers.

Code:-


#include <stdio.h>
int n=5;
int add(int n);
int main()
{
int result;
printf("_____________________________________");
printf("\n\tADD NUMBERS USING THE FUNCTION");
printf("\n_____________________________________\n");

result=add(5);
printf("Result=%d",result);
return 0;
}

int add(int n)
{
int i,sum=0;
for(i=1;i< =n;i++)
{
sum+=i;
}
return sum;
}
