Problem 51

Problem statement:-


Write a C program using global variable, static variable.

Code:-


#include <stdio.h>
void func1();
void func2();
int global_var = 0;

int main()
{
func1();
func1();
return 0;
}

void func1()
{
static int static_var = 0;
global_var++;
static_var++;
printf("\nIn func1, global_var = %d, static_var = %d\n", global_var, static_var);
}

void func2()
{
printf("\nIn func2, global_var = %d\n", global_var);
}
