Problem 18

Problem statement:-


Write a program to reverse a number.

Code:-


#include <stdio.h>
int main()
{
printf("\n________________________________________");
printf("\n\n\tREVERSE THE NUMBER");
printf("\n________________________________________");

int num,rem,rev=0;
printf("\nEnter the number :");
scanf("%d",&num);
while(num>0)
{
rem=num%10;
num=num/10;
rev=rev*10+rem;
}
printf("\nThe reverse number is=%d",rev);
return 0;
}
