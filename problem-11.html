Problem 11

Problem statement:-


Write a C program to check whether the entered character is capital, small letter, digit or any special character.

Code:-


#include <stdio.h>
int main()
{
printf("\n___________________________________________");
printf("\n\tFIND SMALL/CAPITAL CHAR,DIGIT,SPECIAL CHAR");
printf("\n___________________________________________");

char a;
printf("\nEnter the character :");
scanf("%c",&a);

if(a>='A' && a<='Z')
{
printf("\n%c is the Capital character.",a);
}
else if(a>='a' && a<='z')
{
printf("\n%c is the Small character.",a);
}
else if(a>='0' && a<='9')
{
printf("\n%c is the Digit.",a);
}
else
{
printf("\n%c is the Special character.",a);
}
return 0;
}
