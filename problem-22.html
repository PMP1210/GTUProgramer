Problem 22

Problem statement:-


Write a program to calculate average and total of 5 students for 3 subjects. (use nested for loops)

Code:-


#include <stdio.h>
int main()
{
printf("\n_____________________________________________________________");
printf("\n\tTOTAL/AVG. OF 5 STUDENT FOR 3 SUB.");
printf("\n_____________________________________________________________");

int s1[100],s2[100],s3[100],i,j,total[100];
float avg[100];
for(i=1;i<=5;i++)
{
printf("\nEnter the %d student marks of 3 subject:",i);
scanf("\n%d %d %d",&s1[i],&s2[i],&s3[i]);
}
for(i=0;i<=5;i++)
{
for(j=0;j<=5;j++)
{
total[i]=s1[i]+s2[i]+s3[i];
avg[i]=total[i]/3;
}
}
printf("\n_____________________________________________________________");
for(i=1;i<=5;i++)
{
printf("\nThe total of %d student=%d",i,total[i]);
printf("\nThe avg of %d student=%.1f\n",i,avg[i]);
printf("\n_____________________________________________________________");
}
return 0;
}
