Problem 48

Problem statement:-


Write a function Exchange to interchange the values of two variables, say x and y. illustrate the use of this function in a calling function.

Code:-


#include <stdio.h>
void exchange(int* x, int* y)
{
int p=*x;
*x=*y;
*y=p;
}

void main()
{
printf("\n_________________________________________\n");
printf("\tExchange");
printf("\n_________________________________________\n\n");

int x , y;

printf("Enter X : ");
scanf("%d",&x);

printf("\nEnter Y : ");
scanf("%d",&y);

printf("\n------------------------------------\n\n");
printf("Before Exchange: x = %d, y = %d\n", x, y);
exchange(&x, &y);
printf("\nAfter Exchange: x = %d, y = %d\n", x, y);
printf("\n------------------------------------\n\n");
}
