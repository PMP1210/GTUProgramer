Problem 07

Problem statement:-


Write a C program to find out distance travelled by the equation d = vt + at^2

Code:-


#include <stdio.h>
int main()
{
printf("\n________________________________________");
printf("\n\tFIND THE TRAVELL DISTANCE");
printf("\n________________________________________");

float d,u,t,a;
printf("\nEnter the U value :");
scanf("%f",&u);

printf("\nEnter the T value :");
scanf("%f",&t);

printf("\nEnter the A value :");
scanf("%f",&a);

d=(u*t)+(a)*(t*t);

printf("\nThe travell distance is (%.1f * %.1f)+(%.1f * %.1f)^2 = %.2f",u,t,a,t,d);

return 0;
}
