Problem 30

Problem statement:-


Write a program to print following patterns:
*****
****
***
**
*

Code:-


#include <stdio.h>
int main()
{
printf("____________________________________________________________");
printf("\nDIFFERENT TYPES OF STAR PATTEENS");
printf("\n____________________________________________________________");

int n,i,j;
printf("\nEnter the n:");
scanf("%d",&n);
for(i=1;i<=n;i++)
{
for(j=n;j>=i;j--)
{
printf("*");
}
printf("\n");
}
return 0;
}
