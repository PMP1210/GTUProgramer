Problem 39

Problem statement:-


Write a program to find out which number is even or odd from list of 10 numbers using array.

Code:-


#include <stdio.h>
int main()
{
printf("____________________________________________________________");
printf("\nFIND ODD & EVEN FROM 10 NUMBERS VIA ARRAY");
printf("\n____________________________________________________________");

int i,a[100];
printf("\nEnter the 10 numbers:");
for(i=1;i< =10;i++)
{
scanf("%d",&a[i]);
}
for(i=1;i< =10;i++)
{
if(a[i]%2==0)
{
printf("\n%d is even number.",a[i]);
}
else
{
printf("\n%d is odd number.",a[i]);
}
}
return 0;
}
