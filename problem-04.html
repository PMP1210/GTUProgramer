Problem 04

Problem statement:-


Write a C program to interchange two numbers

Code:-


#include stdio.h

int main()
{
printf("\n_____________________");
printf("\n\tSWAPPING");
printf("\n_____________________");
int a,b,c;
printf("\nEnter the A value:");
scanf("%d",&a);

printf("\nEnter the B value:");
scanf("%d",&b);

c=a;
a=b;
b=c;

printf("\n A value:%d",a);
printf("\n\n B value:%d",b);
return 0;
}
