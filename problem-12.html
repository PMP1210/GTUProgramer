Problem 12

Problem statement:-


Write a program to read marks from keyboard and your program should display equivalent gradeaccording to following table. (if else ladder) Marks Grade 100 - 80 Distinction 79 - 60 First Class 59 - 40 Second Class < 40 Fail

Code:-


#include stdio.h
int main()
{
printf("\n________________________________________");
printf("\n\tFIND THE MAXIMUM VALUE");
printf("\n________________________________________");

float a;
printf("\nEnter the marks :");
scanf("%f",&a);

if(a<=100 && a>=80)
{
printf("\nThe result is Distinction.");
}
else if(a<=79 && a>=60)
{
printf("\nThe result is First class.");
}
else if(a<=59 && a>=40)
{
printf("\nThe result is Second class.");
}
else
{
printf("\nSorry,The student is Fail.");
}
return 0;
}
