Problem 40

Problem statement:-


Write a program to find maximum element from 1-Dimensional array.

Code:-


#include <stdio.h>
int main()
{
printf("____________________________________________________________");
printf("\nFIND MAXIMUM ELEMENT FROM 1-D ARRAY");
printf("\n____________________________________________________________");

int i,a[100],n,max=0;
printf("\nEnter the number's of elements:-");
scanf("%d",&n);

for(i=1;i< =n;i++)
{
printf("\nEnter %d element:-",i);
scanf("%d",&a[i]);

if(max< a[i])
{
max=a[i];
}
}

printf("\nThe maximum element is:-%d",max);
return 0;
}
