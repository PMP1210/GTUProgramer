Problem 08

Problem statement:-


Write a C program to find that the accepted number is Negative, or Positive or Zero.

Code:-


#include stdio.h
int main()
{
printf("\n________________________________________");
printf("\n\tFIND THE VALUE IS NEGATIVE , POSSITIVE OR ZERO");
printf("\n________________________________________");

int a;
printf("\nEnter the value :");
scanf("%d",&a);

if(a>0)
{
printf("\nThe %d is possitive.",a);
}
else if(a<0)
{
printf("\nThe %d is negative.",a);
}
else
{
printf("\nThe %d is zero.",a);
}

return 0;
}
