Problem 28

Problem statement:-


Write a program to evaluate the series sum=1-x+x^2/2!-x^3/3!+x^4/4! ..-x^9/9!.

Code:-


#include <stdio.h>
#include <math.h>
float series(float x , int n);
int main()
{
printf("______________________________________________________________________");
printf("\n\t\t\tEVALUATE SERIES SUM=1-x+x^2/2!-x^3/3!+x^4/4!........-x^9/9!");
printf("\n____________________________________________________________________");

float x;
printf("\nEnter the X's value:-");
scanf("%f",&x);

int n=9;
printf("\n %.3f",series(x,n));

return 0;
}

float series(float x , int n)
{
float sum=1,term=1,fac=1,j,y=1,m;
int i;

for(i=1;i< n;i++)
{
for(j=1;j<=y;j++)
{
fac=fac*j;
}
term=pow((-1),i);
m=term*pow(x,y)/fac;
sum=sum+m;
y++;
}
return sum;
}
