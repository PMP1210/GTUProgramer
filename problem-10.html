Problem 10

Problem statement:-


Write a program to read three numbers from keyboard and find out maximum out of these three. (Nested if else)

Code:-


#include stdio.h
int main()
{
printf("\n________________________________________");
printf("\n\tFIND THE MAXIMUM VALUE");
printf("\n________________________________________");

float a,b,c;
printf("\nEnter the three values :");
scanf("%f %f %f",&a,&b,&c);

if(a>b && a>c)
{
printf("\nThe %.1f is greater than %.1f and %.1f.",a,b,c);
}
else if(b>a && b>c)
{
printf("\nThe %.1f is greater than %.1f and %.1f.",b,a,c);
}
else if(c>a && c>b)
{
printf("\nThe %.1f is greater than %.1f and %.1f.",c,a,b);
}

return 0;

}
