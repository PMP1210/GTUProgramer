Problem 47

Problem statement:-


Write a function in the program to return 1 if number is prime otherwise return 0.

Code:-


#include <stdio.h>
int prime(int n)
{
int i;
if (n < = 1)
{
return 0;
}
for(i=2;i< n;i++)
{
if (n%i==0)
{
return 0;
}
}
return 1;
}

void main()
{
int n;
printf("\n_________________________________________\n");
printf("\tPrime Number");
printf("\n_________________________________________\n\n");

printf("Enter The Number : ");
scanf("%d",&n);
int result = prime(n);

printf("\n------------------------------------\n\n");

if (result == 1)
{
printf("%d is a prime number\n", n);
} else
{
printf("%d is not a prime number\n", n);
}
printf("\n------------------------------------\n\n");
}
