Problem 14

Problem statement:-


Write a C program to read no 1 to 7 and print relatively day Sunday to Saturday.

Code:-


#include <stdio.h>
int main()
{
printf("\n________________________________________");
printf("\n\tDAYS OF WEEK");
printf("\n________________________________________");

int a;
printf("\nEnter the number :");
scanf("%d",&a);

if(a==1)
{
printf("\nToday is Sunday.");
}
else if(a==2)
{
printf("\nToday is Monday.");
}
else if(a==3)
{
printf("\nToday is Tuesday.");
}
else if(a==4)
{
printf("\nToday is Wednesday.");
}
else if(a==5)
{
printf("\nToday is Thursday.");
}
else if(a==6)
{
printf("\nToday is Friday.");
}
else if(a==7)
{
printf("\nToday is Saturday.");
}
return 0;
}
