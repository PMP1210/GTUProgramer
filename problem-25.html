Problem 25

Problem statement:-


Write a program to evaluate the series 1^2+2^2+3^2+......+n^2.

Code:-


#include <stdio.h>
int main()
{
printf("\n_____________________________________________________________");
printf("\n\tEVALUATE THE SERIES 1^2+2^2+3^2+......+n^2");
printf("\n_____________________________________________________________");

int n,i,sum=0;
printf("\nEnter the n value :");
scanf("%d",&n);

for(i=1;i<=n;i++)
{
sum=sum+(i*i);
}
printf("Sum of searies = %d",sum);
return 0;
}
